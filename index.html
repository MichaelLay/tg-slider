How to use

1. Load jQuery and include TG Slider plugin files

<!-- Animation stylesheet -->
<link type="text/css" rel="stylesheet" href="css/animate.css">
<!-- TG Slider stylesheet -->
<link type="text/css" rel="stylesheet" href="css/tg-slider.min.css">
<!-- FontAwesome Icon -->
<link rel="stylesheet" href="css/font-awesome.min.css">
<!--  jQuery -->
<script type="text/javascript" src="js/jquery.min.js"></script>
<!-- TG Slider script -->
<script type="text/javascript" src="js/tg-slider.min.js"></script>

2. Setup your HTML

Need to be exactly like this: 

<div id="tg-slider-wrapper">
    <ul class="tg-slider">
        <li>
            <img src="Path to your image" alt="" title=""/>
        </li>
        <li>
            <img src="Path to your image" alt="" title=""/>
        </li>
        <li>
            <img src="Path to your image" alt="" title=""/>
        </li>
    </ul>
    <ul id="tg-nav">
        <li>
            <a class="tg-nav prev"><i class="fa fa-angle-left"> </i></a>
        </li>
        <li class="align-right">
            <a class="tg-nav next"><i class="fa fa-angle-right"> </i></a>
        </li>
    </ul>
</div>

3. Call TG Slider plugin

$(document).ready(function() {
    $("ul.tg-slider").tgSlider();
});

4. Set options

$(document).ready(function() {
    $("ul.tg-slider").tgSlider({
        speed: 1000, // Slide speed, default 3000
        height: 600, // Set slider container height, default 600
    });
});
